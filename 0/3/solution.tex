	\begin{cjsection}{Simplify the following. Rationalize the denominators.}
		\cjitem{\(\frac{24c^{-\frac{1}{2}}d^{\frac{2}{3}}}{18c^{-\frac{1}{7}}d^{-\frac{3}{5}}}\)}
			\begin{cjsolution}
				\cjwhy{\(\frac{4c^{-\frac{1}{2}}d^{\frac{2}{3}}}{3c^{-\frac{1}{7}}d^{-\frac{3}{5}}}\)}{Simplify the fraction to lowest terms.}
				\cjwhy{\(\frac{4d^{\frac{2}{3}}c^{\frac{1}{7}}d^{\frac{3}{5}}}{3c^{\frac{1}{2}}}\)}{\(a^{-\frac{b}{c}} = \frac{1}{a^\frac{b}{c}}\)}
				\cjwhy{\(\frac{4d^{\frac{2}{3}}d^{\frac{3}{5}}}{3}c^{\frac{1}{7} - \frac{1}{2}}\)}{\(\frac{a^m}{a^n} = a^{m - n}\)}
				\cjwhy{\(\frac{4d^{\frac{2}{3}}d^{\frac{3}{5}}}{3}c^{\frac{2}{14} - \frac{7}{14}}\)}{\(\text{LCM} = 14\)}
				\cjcontinue{\(\frac{4d^{\frac{2}{3}}d^{\frac{3}{5}}}{3}c^{-\frac{5}{14}}\)}
				\cjwhy{\(\frac{4}{3}c^{-\frac{5}{14}}d^{\frac{2}{3} + \frac{3}{5}}\)}{\(a^ma^n = a^{m + n}\)}
				\cjwhy{\(\frac{4}{3}c^{-\frac{5}{14}}d^{\frac{10}{15} + \frac{9}{15}}\)}{\(\text{LCM} = 15\)}
				\cjcontinue{\(\frac{4}{3}c^{-\frac{5}{14}}d^{\frac{19}{15}}\)}
				\cjwhy{\(\frac{4d^{\frac{19}{15}}}{3c^{\frac{5}{14}}}\)}{\(a^{-\frac{b}{c}} = \frac{1}{a^\frac{b}{c}}\)}
				\cjcontinue{\(\frac{4\sqrt[15]{d^{19}}}{3\sqrt[14]{c^5}}\)}
				\cjwhy{\(\frac{4\sqrt[15]{d^{19}}}{3\sqrt[14]{c^5}} \cdot \frac{\sqrt[14]{c^9}}{\sqrt[14]{c^9}}\)}{Rationalize.}
				\cjcontinue{\(\frac{4\sqrt[14]{c^9}\sqrt[15]{d^{19}}}{3c}\)}
			\end{cjsolution}

		\cjitem{\((u^{\frac{1}{3}} + {(uv)}^{\frac{1}{6}} + v^{\frac{1}{3}})(u^{\frac{1}{6}} - v^{\frac{1}{6}})\)}
			\begin{cjsolution}
				\cjwhy{\((u^{\frac{1}{3}} + u^{\frac{1}{6}}v^{\frac{1}{6}} + v^{\frac{1}{3}})(u^{\frac{1}{6}} - v^{\frac{1}{6}})\)}{Distribute exponent.}
				\cjwhy{\(u^{\frac{1}{2}} - v^{\frac{1}{2}}\)}{Use difference of two cubes.}
				\cjcontinue{\(\sqrt{u} - \sqrt{v}\)}
			\end{cjsolution}

		\cjitem{\(\sqrt[3]{-8^4}\)}
			\begin{cjsolution}
				\cjwhy{\(-\sqrt[3]{8^4}\)}{\(\sqrt[m]{-a} = -\sqrt[m]{a} \text{ for odd } m\)}
				\cjcontinue{\(-\sqrt[3]{{(2^3)}^4}\)}
				\cjwhy{\(-\sqrt[3]{{(2^4)}^3}\)}{\({(a^m)}^n = {(a^n)}^m\)}
				\cjcontinue{\(-2^4\)}
				\cjcontinue{\(-16\)}
			\end{cjsolution}

		\cjitem{\(\sqrt[4]{9x^8}\)}
			\begin{cjsolution}
				\cjwhy{\(\sqrt[4]{9}\sqrt[4]{x^8}\)}{\(\sqrt[m]{ab} = \sqrt[m]{a} \sqrt[m]{b}\)}
				\cjcontinue{\(\sqrt[4]{3^2}\sqrt[4]{x^8}\)}
				\cjcontinue{\(x^2\sqrt{3}\)}
			\end{cjsolution}

		\cjitem{\(\sqrt{\sqrt[3]{9a^4b^4}}\)}
			\begin{cjsolution}
				\cjwhy{\(\sqrt[6]{9a^4b^4}\)}{\(\sqrt[m]{\sqrt[n]{a}} = \sqrt[m + n]{a}\)}
				\cjcontinue{\(\sqrt[6]{3^2a^4b^4}\)}
				\cjcontinue{\(\sqrt[3]{3a^2b^2}\)}
			\end{cjsolution}

		\cjitem{\(\frac{2\sqrt{5}}{\sqrt{8}} + \frac{9}{\sqrt[3]{16}}\)}
			\begin{cjsolution}
				\cjwhy{\(\frac{2\sqrt{5}}{\sqrt{8}} \cdot \frac{\sqrt{2}}{\sqrt{2}} + \frac{9}{\sqrt[3]{16}} \cdot \frac{\sqrt[3]{4}}{\sqrt[3]{4}}\)}{Rationalize.}
				\cjcontinue{\(\frac{2\sqrt{5}\sqrt{2}}{\sqrt{16}} + \frac{9\sqrt[3]{4}}{\sqrt[3]{64}}\)}
				\cjcontinue{\(\frac{2\sqrt{5}\sqrt{2}}{4} + \frac{9\sqrt[3]{4}}{4}\)}
				\cjcontinue{\(\frac{2\sqrt{5}\sqrt{2} + 9\sqrt[3]{4}}{4}\)}
				\cjcontinue{\(\frac{2\sqrt{10} + 9\sqrt[3]{4}}{4}\)}
			\end{cjsolution}

		\cjitem{\(\frac{x^2 - 2x + 1}{\sqrt{x} + 1}\)}
			\begin{cjsolution}
				\cjwhy{\(\frac{x^2 - 2x + 1}{\sqrt{x} + 1} \cdot \frac{\sqrt{x} - 1}{\sqrt{x} - 1}\)}{Rationalize using difference of two squares.}
				\cjcontinue{\(\frac{(x^2 - 2x + 1)(\sqrt{x} - 1)}{x - 1}\)}
				\cjwhy{\(\frac{{(x - 1)}^2(\sqrt{x} - 1)}{x - 1}\)}{Factor by grouping.}
				\cjcontinue{\((x - 1)(\sqrt{x} - 1)\)}
			\end{cjsolution}

		\cjitem{\(\frac{1}{\sqrt[3]{4} - \sqrt[3]{-27}}\)}
			\begin{cjsolution}
				\cjwhy{\(\frac{1}{\sqrt[3]{4} + \sqrt[3]{27}}\)}{\(\sqrt[m]{-a} = -\sqrt[m]{a} \text{ for odd } m\)}
				\cjwhy{\(\frac{1}{\sqrt[3]{4} + \sqrt[3]{27}} \cdot \frac{\sqrt[3]{4^2} - \sqrt[3]{4}\sqrt[3]{27} + \sqrt[3]{27^2}}{\sqrt[3]{4^2} - \sqrt[3]{4}\sqrt[3]{27} + \sqrt[3]{27^2}}\)}{Rationalize using difference of two cubes.}
				\cjcontinue{\(\frac{\sqrt[3]{4^2} - \sqrt[3]{4}\sqrt[3]{27} + \sqrt[3]{27^2}}{4 + 27}\)}
				\cjcontinue{\(\frac{\sqrt[3]{4^2} - 3\sqrt[3]{4} + \sqrt[3]{27^2}}{4 + 27}\)}
				\cjcontinue{\(\frac{\sqrt[3]{4^2} - 3\sqrt[3]{4} + \sqrt[3]{27^2}}{31}\)}
				\cjcontinue{\(\frac{\sqrt[3]{16} - 3\sqrt[3]{4} + \sqrt[3]{27^2}}{31}\)}
				\cjcontinue{\(\frac{\sqrt[3]{16} - 3\sqrt[3]{4} + \sqrt[3]{{(3^3)}^2}}{31}\)}
				\cjwhy{\(\frac{\sqrt[3]{16} - 3\sqrt[3]{4} + \sqrt[3]{{(3^2)}^3}}{31}\)}{\({(a^m)}^n = {(a^n)}^m\)}
				\cjcontinue{\(\frac{\sqrt[3]{16} - 3\sqrt[3]{4} + 3^2}{31}\)}
				\cjcontinue{\(\frac{\sqrt[3]{16} - 3\sqrt[3]{4} + 9}{31}\)}
				\cjwhy{\(\frac{\sqrt[3]{8}\sqrt[3]{2} - 3\sqrt[3]{4} + 9}{31}\)}{\(\sqrt[m]{ab} = \sqrt[m]{a} \sqrt[m]{b}\)}
				\cjcontinue{\(\frac{2\sqrt[3]{2} - 3\sqrt[3]{4} + 9}{31}\)}
			\end{cjsolution}
	\end{cjsection}



	\begin{cjsection}{Perform the following operations and simplify.}
		\cjitem{\(3i(i^2 - i^3 + 5i^5 - i^{-2})\)}
			\begin{cjsolution}
				\cjwhy{\(3i(-1 - i^3 + 5i^5 - i^{-2})\)}{\(i^2 = -1\)}
				\cjwhy{\(3i(-1 + i + 5i^5 - i^{-2})\)}{\(i^3 = -i\)}
				\cjwhy{\(3i(-1 + i + 5i - i^{-2})\)}{\(i^5 = i\)}
				\cjwhy{\(3i(-1 + i + 5i + 1)\)}{\(i^{-2} = -1\)}
				\cjcontinue{\(3i(6i)\)}
				\cjcontinue{\(18i^2\)}
				\cjwhy{\(-18\)}{\(i^2 = -1\)}
			\end{cjsolution}

		\cjitem{\((3 - 5i)(7 + 4i)\)}
			\begin{cjsolution}
				\cjwhy{\(21 + 12i - 35i - 20i^2\)}{Expand.}
				\cjwhy{\(21 + 12i - 35i + 20\)}{\(i^2 = -1\)}
				\cjcontinue{\(41 - 23i\)}
			\end{cjsolution}
	\end{cjsection}
