	\cjboilerplate{Math 20}{Factoring Polynomials and Simplifying Rational Expressions}

	\begin{cjsection}{Factor the following completely.}
		\cjitem{\( 16x^4 - 1 \)}
			\begin{cjsolution}
				\cjwhy{\((4x^2 - 1)(4x^2 + 1)\)}{Factor using difference of two squares.}
				\cjwhy{\((2x - 1)(2x + 1)(4x^2 + 1)\)}{Factor using difference of two squares.}
			\end{cjsolution}

		\cjitem{\(8j^3 - 125k^6\)}
			\begin{cjsolution}
				\cjwhy{\((2j - 5k^2)(4j^2 + 10jk^2 + 25k^4)\)}{Factor using difference of two cubes.}
			\end{cjsolution}

		\cjitem{\(s^2 + 7s + 10\)}
			\begin{cjsolution}
				\cjwhy{\((s + 2)(s + 5)\)}{Factor by grouping.}
			\end{cjsolution}

		\cjitem{\(4n^2 - 12n + 9\)}
			\begin{cjsolution}
				\cjwhy{\(4n^2 - 6n - 6n + 9\)}{Factor by grouping.}
				\cjcontinue{\(2n(2n - 3) - 3(2n - 3)\)}
				\cjcontinue{\({(2n - 3)}^2\)}
			\end{cjsolution}

		\cjitem{\(x^3 - x^2 - x + 1\)}
			\begin{cjsolution}
				\cjwhy{\(x^2(x - 1) - 1(x - 1)\)}{Factor by grouping.}
				\cjcontinue{\((x^2 - 1)(x - 1)\)}
				\cjwhy{\((x - 1)(x + 1)(x - 1)\)}{Factor using difference of two squares.}
				\cjcontinue{\({(x - 1)}^2(x + 1)\)}
			\end{cjsolution}

		\cjitem{\(48 - 13q - q^2\)}
			\begin{cjsolution}
				\cjwhy{\(-q^2 - 13q + 48\)}{Rewrite in standard form.}
				\cjcontinue{\(-(q^2 + 13q - 48)\)}
				\cjwhy{\(-(q - 3)(q + 16)\)}{Factor by grouping.}
			\end{cjsolution}
	\end{cjsection}



	\begin{cjsection}{Reduce the following rational expressions to lowest terms.}
		\cjitem{\(\frac{a^2 - b^2}{a^3 - b^3}\)}
			\begin{cjsolution}
				\cjwhy{\(\frac{(a - b)(a + b)}{a^3 - b^3}\)}{Factor using difference of two squares.}
				\cjwhy{\(\frac{(a - b)(a + b)}{(a - b)(a^2 + ab + b^2)}\)}{Factor using difference of two cubes.}
				\cjcontinue{\(\frac{a + b}{a^2 + ab + b^2}\)}
			\end{cjsolution}

		\cjitem{\(\frac{x^3 - x^2y + xy^2 - y^3}{x^6 + y^6}\)}
			\begin{cjsolution}
				\cjwhy{\(\frac{x^2(x - y) + y^2(x - y)}{x^6 + y^6}\)}{Factor by grouping.}
				\cjcontinue{\(\frac{(x^2 + y^2)(x - y)}{x^6 + y^6}\)}
				\cjwhy{\(\frac{(x^2 + y^2)(x - y)}{(x^2 + y^2)(x^4 - x^2y^2 + y^4)}\)}{Factor using difference of two cubes.}
				\cjcontinue{\(\frac{x - y}{x^4 - x^2y^2 + y^4}\)}
			\end{cjsolution}
	\end{cjsection}
