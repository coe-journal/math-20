	\cjboilerplate{Math 20}{Equations in Quadratic Form and with Radicals and Absolute Values}

	\begin{cjsection}{Find the solution set of the following inequalities.}
		\cjitem{\(\frac{2x + 1}{4} \le \frac{2x}{3} + \frac{1}{6}\)}
			\begin{cjsolution}
				\cjwhy{\(\frac{3(2x + 1)}{12} \le \frac{4(2x)}{12} + \frac{2}{12}\)}{LCM = \(12\)}
				\cjcontinue{\(\frac{6x + 3}{12} \le \frac{8x}{12} + \frac{2}{12}\)}
				\cjcontinue{\(\frac{6x + 3}{12} \le \frac{8x + 2}{12}\)}
				\cjcontinue{\(6x + 3 \le 8x + 2\)}
				\cjcontinue{\(3 - 2 \le 8x - 6x\)}
				\cjcontinue{\(1 \le 2x\)}
				\cjcontinue{\(x \ge \frac{1}{2}\)}
				\cjfa{\(x \in [\frac{1}{2}, +\infty)\)}
			\end{cjsolution}

		\cjitem{\(-2 < 5 + 3x < 20\)}
			\begin{cjsolution}
				\cjwhy{\(-7 < 3x < 15\)}{Solve for \(x\).}
				\cjcontinue{\(-\frac{7}{3} < x < 5\)}
				\cjfa{\(x \in (-\frac{7}{3}, 5)\)}
			\end{cjsolution}

		\cjitem{\(\frac{x}{x - 1} > -1\)}
			\begin{cjsolution}
				\cjwhy{\(\frac{x}{x - 1} + 1 > 0\)}{Solve for \(x\).}
				\cjcontinue{\(\frac{x}{x - 1} + \frac{x - 1}{x - 1} > 0\)}
				\cjcontinue{\(\frac{x + x - 1}{x - 1} > 0\)}
				\cjsubwhy{\(\frac{2x - 1}{x - 1} > 0\)}{\(x = 1\) is an undefined point.}
				\cjsign{
					\tkzTabInit[lgt=2, espcl=2, deltacl=0]
						{/.8, \(2x - 1\) /.8, \(x - 1\) /.8, \(\frac{2x - 1}{x - 1}\) /.8}
						{,\(\frac{1}{2}\),\(1\),}
					\tkzTabLine{,-,t,+,t,+,}
					\tkzTabLine{,-,t,-,t,+,}
					\tkzTabLine{,+,t,-,t,+,}
				}
				\cjfa{\(x \in (-\infty, \frac{1}{2}) \cup (1, +\infty)\)}
			\end{cjsolution}

		\cjitem{\(\frac{x}{x + 1} \ge \frac{2}{x + 3}\)}
			\begin{cjsolution}
				\cjwhy{\(\frac{x}{x + 1} - \frac{2}{x + 3} \ge 0\)}{Solve for \(x\).}
				\cjsubwhy{\(\frac{x(x + 3)}{(x + 1)(x + 3)} - \frac{2(x + 1)}{(x + 1)(x + 3)} \ge 0\)}{LCM = \((x + 1)(x + 3)\)}
				\cjcontinue{\(\frac{x(x + 3) - 2(x + 1)}{(x + 1)(x + 3)} \ge 0\)}
				\cjcontinue{\(\frac{x^2 + 3x - 2x - 2}{(x + 1)(x + 3)} \ge 0\)}
				\cjcontinue{\(\frac{x^2 + x - 2}{(x + 1)(x + 3)} \ge 0\)}
				\cjsubwhy{\(\frac{(x - 1)(x + 2)}{(x + 1)(x + 3)} \ge 0\)}{Factor by grouping. \(x \in \{-3, -1\}\) are undefined points.}
				\cjsign{
					\tkzTabInit[lgt=2, espcl=2, deltacl=0]
						{/.8, \(x - 1\) /.8, \(x + 2\) /.8, \(x + 1\) /.8, \(x + 3\) /.8, \(\frac{(x - 1)(x + 2)}{(x + 1)(x + 3)}\) /.8}
						{,\(-3\),\(-2\),\(-1\),\(1\),}
					\tkzTabLine{,-,t,-,t,-,t,-,t,+}
					\tkzTabLine{,-,t,-,t,+,t,+,t,+}
					\tkzTabLine{,-,t,-,t,-,t,+,t,+}
					\tkzTabLine{,-,t,+,t,+,t,+,t,+}
					\tkzTabLine{,+,t,-,t,+,t,-,t,+}
				}
				\cjfastep{\((-\infty, -3) \cup [-2, -1) \cup [1, +\infty)\)}{Don't include undefined points.}
			\end{cjsolution}

		\cjitem{\(\abs{\frac{9 - 2x}{4x}} \ge 1\)}
			\begin{cjsolution}
				\cjwhy{\(\frac{9 - 2x}{4x} \ge 1\)}{\(\abs{a} \ge b \Rightarrow a \ge b\) or \(a \le -b\). Solve for \(a \ge b\).}
				\cjcontinue{\(\frac{9 - 2x}{4x} - 1 \ge 0\)}
				\cjcontinue{\(\frac{9 - 2x}{4x} - \frac{4x}{4x} \ge 0\)}
				\cjcontinue{\(\frac{9 - 2x - 4x}{4x} \ge 0\)}
				\cjcontinue{\(\frac{9 - 6x}{4x} \ge 0\)}
				\cjcontinue{\(\frac{9 - 6x}{x} \ge 0\)}
				\cjcontinue{\(\frac{-3(2x - 3)}{x} \ge 0\)}
				\cjsubwhy{\(\frac{2x - 3}{x} \le 0\)}{\(x = 0\) is an undefined point.}
				\cjsign{
					\tkzTabInit[lgt=2, espcl=2, deltacl=0]
						{/.8, \(2x - 3\) /.8, \(x\) /.8, \(\frac{2x - 3}{x}\) /.8}
						{,\(0\),\(\frac{3}{2}\),}
					\tkzTabLine{,-,t,-,t,+,}
					\tkzTabLine{,-,t,+,t,+,}
					\tkzTabLine{,+,t,-,t,+,}
				}
				\cjcontinue{\(x \in (0, \frac{3}{2}]\)}
				\cjwhy{\(\frac{9 - 2x}{4x} \le -1\)}{\(\abs{a} \ge b \Rightarrow a \ge b\) or \(a \le -b\). Solve for \(a \le -b\).}
				\cjcontinue{\(\frac{9 - 2x}{4x} + 1 \le 0\)}
				\cjcontinue{\(\frac{9 - 2x}{4x} + \frac{4x}{4x} \le 0\)}
				\cjcontinue{\(\frac{9 - 2x + 4x}{4x} \le 0\)}
				\cjcontinue{\(\frac{2x + 9}{4x} \le 0\)}
				\cjsubwhy{\(\frac{2x + 9}{x} \le 0\)}{\(x = 0\) is an undefined point.}
				\cjsign{
					\tkzTabInit[lgt=2, espcl=2, deltacl=0]
						{/.8, \(2x + 9\) /.8, \(x\) /.8, \(\frac{2x + 9}{x}\) /.8}
						{,\(-\frac{9}{2}\),\(0\),}
					\tkzTabLine{,-,t,+,t,+,}
					\tkzTabLine{,-,t,-,t,+,}
					\tkzTabLine{,+,t,-,t,+,}
				}
				\cjcontinue{\(x \in [-\frac{9}{2}, 0)\)}
				\cjfastep{\(x \in [-\frac{9}{2}, 0) \cup (0, \frac{3}{2}]\)}{Combine intervals.}
			\end{cjsolution}

		\cjitem{\(\abs{\frac{x}{2x - 3}} \le 1\)}
			\begin{cjsolution}
				\cjwhy{\(\frac{x}{2x - 3} \le 1\)}{\(\abs{a} \le b \Rightarrow a \le b\) and \(a \ge -b\). Solve for \(a \le b\).}
				\cjcontinue{\(\frac{x}{2x - 3} - 1 \le 0\)}
				\cjcontinue{\(\frac{x}{2x - 3} - \frac{2x - 3}{2x - 3} \le 0\)}
				\cjcontinue{\(\frac{x - (2x - 3)}{2x - 3} \le 0\)}
				\cjcontinue{\(\frac{x - 2x + 3}{2x - 3} \le 0\)}
				\cjcontinue{\(\frac{-x + 3}{2x - 3} \le 0\)}
				\cjcontinue{\(\frac{-(x - 3)}{2x - 3} \le 0\)}
				\cjsubwhy{\(\frac{x - 3}{2x - 3} \ge 0\)}{\(x = \frac{3}{2}\) is an undefined point.}
				\cjsign{
					\tkzTabInit[lgt=2, espcl=2, deltacl=0]
						{/.8, \(x - 3\) /.8, \(2x - 3\) /.8, \(\frac{x - 3}{2x - 3}\) /.8}
						{,\(\frac{3}{2}\),\(3\),}
					\tkzTabLine{,-,t,-,t,+,}
					\tkzTabLine{,-,t,+,t,+,}
					\tkzTabLine{,+,t,-,t,+,}
				}
				\cjcontinue{\(x \in (-\infty, \frac{3}{2}) \cup [3, +\infty)\)}
				\cjwhy{\(\frac{x}{2x - 3} \ge -1\)}{\(\abs{a} \le b \Rightarrow a \le b\) and \(a \ge -b\). Solve for \(a \ge -b\).}
				\cjcontinue{\(\frac{x}{2x - 3} + 1 \ge 0\)}
				\cjcontinue{\(\frac{x}{2x - 3} + \frac{2x - 3}{2x - 3} \ge 0\)}
				\cjcontinue{\(\frac{x + 2x - 3}{2x - 3} \ge 0\)}
				\cjcontinue{\(\frac{3x - 3}{2x - 3} \ge 0\)}
				\cjcontinue{\(\frac{3(x - 1)}{2x - 3} \ge 0\)}
				\cjsubwhy{\(\frac{x - 1}{2x - 3} \ge 0\)}{\(x = \frac{3}{2}\) is an undefined point.}
				\cjsign{
					\tkzTabInit[lgt=2, espcl=2, deltacl=0]
						{/.8, \(x - 1\) /.8, \(2x - 3\) /.8, \(\frac{x - 1}{2x - 3}\) /.8}
						{,\(1\),\(\frac{3}{2}\),}
					\tkzTabLine{,-,t,+,t,+,}
					\tkzTabLine{,-,t,-,t,+,}
					\tkzTabLine{,+,t,-,t,+,}
				}
				\cjcontinue{\(x \in (-\infty, 1] \cup (\frac{3}{2}, +\infty)\)}
				\cjwhy{\(x \in ((-\infty, \frac{3}{2}) \cup [3, +\infty)) \cap ((-\infty, 1] \cup (\frac{3}{2}, +\infty))\)}{Combine intervals.}
				\cjfa{\(x \in (-\infty, 1] \cup [3, +\infty)\)}
			\end{cjsolution}

		\cjitem{\(0 < \abs{x - 5} < 2\)}
			\begin{cjsolution}
				\cjwhy{\(\abs{x - 5} > 0, \abs{x - 5} < 2\)}{Split the inequality.}
				\cjwhy{\(x - 5 > 0\)}{\(\abs{a} > b \Rightarrow a > b\) or \(a < -b\). Solve for \(a > b\).}
				\cjcontinue{\(x > 5\)}
				\cjcontinue{\(x \in (5, +\infty)\)}
				\cjwhy{\(x - 5 < 0\)}{\(\abs{a} > b \Rightarrow a > b\) or \(a < -b\). Solve for \(a < -b\).}
				\cjcontinue{\(x < 5\)}
				\cjcontinue{\(x \in (-\infty, 5)\)}
				\cjwhy{\(x - 5 < 2\)}{\(\abs{a} < b \Rightarrow a < b\) and \(a > -b\). Solve for \(a < b\).}
				\cjcontinue{\(x < 7\)}
				\cjcontinue{\(x \in (-\infty, 7)\)}
				\cjwhy{\(x - 5 > -2\)}{\(\abs{a} < b \Rightarrow a < b\) and \(a > -b\). Solve for \(a > -b\).}
				\cjcontinue{\(x > 3\)}
				\cjcontinue{\(x \in (3, +\infty)\)}
				\cjwhy{\(x \in ((-\infty, 5) \cup (5, +\infty)) \cap ((-\infty, 7) \cap (3, +\infty))\)}{Combine intervals.}
				\cjcontinue{\(x \in ((-\infty, 5) \cup (5, +\infty)) \cap (3, 7)\)}
				\cjfa{\(x \in (3, 5) \cup (5, 7)\)}
			\end{cjsolution}
	\end{cjsection}
