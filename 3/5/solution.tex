	\begin{cjsection}{Evaluate the following without using a calculator.}
		\cjitem{\(\sin(\frac{19\pi}{12})\)}
			\begin{cjsolution}
				\cjwhy{\(\sin(\frac{10\pi}{12} + \frac{9\pi}{12})\)}{}
				\cjcontinue{\(\sin(\frac{5\pi}{6} + \frac{3\pi}{4})\)}
				\cjsubwhy{\(\sin(\frac{5\pi}{6})\cos(\frac{3\pi}{4}) + \cos(\frac{5\pi}{6})\sin(\frac{3\pi}{4})\)}{\(\sin(a + b) = \sin(a)\cos(b) + \cos(a)\sin(b)\)}
				\cjcontinue{\((\frac{1}{2})(-\frac{\sqrt{2}}{2}) + (-\frac{\sqrt{3}}{2})(\frac{\sqrt{2}}{2})\)}
				\cjcontinue{\(-\frac{\sqrt{2}}{4} - \frac{\sqrt{3}\sqrt{2}}{4}\)}
				\cjcontinue{\(-\frac{\sqrt{2}}{4} - \frac{\sqrt{6}}{4}\)}
				\cjfa{\(-\frac{\sqrt{2} + \sqrt{6}}{4}\)}
			\end{cjsolution}

		\cjitem{\(\cos(33\degree)\cos(27\degree) - \sin(33\degree)\sin(27\degree)\)}
			\begin{cjsolution}
				\cjwhy{\(\cos(33\degree + 27\degree)\)}{\(\cos(a + b) = \cos(a)\cos(b) - \sin(a)\sin(b)\)}
				\cjcontinue{\(\cos(60\degree)\)}
				\cjfa{\(\frac{1}{2}\)}
			\end{cjsolution}
	\end{cjsection}



	\begin{cjsection}{If \(\cot(\theta) = -\frac{5}{12}\) and \(\theta \in (-\frac{\pi}{2}, 0)\), find \(\cos(\theta + \frac{\pi}{3})\).}
		\begin{cjsolution}
			\cjwhy{\(O = -12, A = 5\)}{\(\cot(\theta) = \frac{A}{O}\), and since \(\theta \in (-\frac{\pi}{2}, 0)\), we are in QIV. Therefore, \(O < 0\) and \(A > 0\).}
			\cjsubwhy{\(H = \sqrt{{(-12)}^2 + 5^2}\)}{\(H = \sqrt{O^2 + A^2}\)}
			\cjcontinue{\(H = \sqrt{144 + 25}\)}
			\cjcontinue{\(H = \sqrt{169}\)}
			\cjcontinue{\(H = 13\)}
			\cjwhy{\(\cos(\theta + \frac{\pi}{3}) = \cos(\theta)\cos(\frac{\pi}{3}) - \sin(\theta)\sin(\frac{\pi}{3})\)}{\(\cos(a + b) = \cos(a)\cos(b) - \sin(a)\sin(b)\)}
			\cjsubwhy{\(\cos(\theta + \frac{\pi}{3}) = \frac{5}{13}\cos(\frac{\pi}{3}) - \sin(\theta)\sin(\frac{\pi}{3})\)}{\(\cos(\theta) = \frac{A}{H}\)}
			\cjsubwhy{\(\cos(\theta + \frac{\pi}{3}) = \frac{5}{13}\cos(\frac{\pi}{3}) + \frac{12}{13}\sin(\frac{\pi}{3})\)}{\(\sin(\theta) = \frac{O}{H}\)}
			\cjcontinue{\(\cos(\theta + \frac{\pi}{3}) = (\frac{5}{13})(\frac{1}{2}) + (\frac{12}{13})(\frac{\sqrt{3}}{2})\)}
			\cjcontinue{\(\cos(\theta + \frac{\pi}{3}) = \frac{5}{26} + \frac{12\sqrt{3}}{26}\)}
			\cjcontinue{\(\cos(\theta + \frac{\pi}{3}) = \frac{5 + 12\sqrt{3}}{26}\)}
		\end{cjsolution}
	\end{cjsection}



	\begin{cjsection}{Given \(\cos(\alpha) = \frac{3}{5}\) where \(\alpha\) lies in the interval \((\frac{\pi}{2}, 2\pi)\), find the following.}
		\cjitem{\(\sin(2\alpha)\)}
			\begin{cjsolution}
				\cjwhy{\(A = 3, H = 5\)}{\(\cos(\theta) = \frac{A}{H}\), and since \(\cos(\alpha) > 0\) and \(\alpha \in (\frac{\pi}{2}, 2\pi)\), we are in QIV. Therefore, \(O < 0\) and \(A > 0\).}
				\cjsubwhy{\(O = -\sqrt{5^2 - 3^2}\)}{From \(H = \sqrt{A^2 + O^2}\), we can derive \(O = \pm \sqrt{H^2 - A^2}\). Remember that in this case, \(A > 0\).}
				\cjcontinue{\(O = -\sqrt{25 - 9}\)}
				\cjcontinue{\(O = -\sqrt{16}\)}
				\cjcontinue{\(O = -4\)}
				\cjsubwhy{\(\sin(\alpha) = -\frac{4}{5}\)}{\(\sin(\theta) = \frac{O}{H}\)}
				\cjwhy{\(\sin(2\alpha) = 2(-\frac{4}{5})(\frac{3}{5})\)}{\(\sin(2\theta) = 2\sin(\theta)\cos(\theta)\)}
				\cjcontinue{\(\sin(2\alpha) = 2(-\frac{12}{25})\)}
				\cjfa{\(\sin(2\alpha) = -\frac{24}{25}\)}
			\end{cjsolution}

		\cjitem{\(\sin(3\alpha)\)}
			\begin{cjsolution}
				\cjwhy{\(\cos(2\alpha) = {(\frac{3}{5})}^2 - {(-\frac{4}{5})^2}\)}{\(\cos(2\theta) = \cos^2(\theta) - \sin^2(\theta)\)}
				\cjcontinue{\(\cos(2\alpha) = \frac{9}{25} - \frac{16}{25}\)}
				\cjcontinue{\(\cos(2\alpha) = -\frac{7}{25}\)}
				\cjwhy{\(\sin(3\alpha) = \sin(\alpha + 2\alpha)\)}{}
				\cjsubwhy{\(\sin(\alpha + 2\alpha) = (-\frac{4}{5})(-\frac{7}{25}) + (\frac{3}{5})(-\frac{24}{25})\)}{\(\sin(a + b) = \sin(a)\cos(b) + \cos(a)\sin(b)\)}
				\cjcontinue{\(\sin(\alpha + 2\alpha) = \frac{28}{125} - \frac{72}{125}\)}
				\cjcontinue{\(\sin(\alpha + 2\alpha) = \frac{44}{125}\)}
				\cjfa{\(\sin(3\alpha) = \frac{44}{125}\)}
			\end{cjsolution}
	\end{cjsection}



	\begin{cjsection}{Establish the following identities.}
		\cjitem{\(\frac{\sin(\alpha - \beta)}{\cos(\alpha)\cos(\beta)} = \tan(\alpha) - \tan(\beta)\)}
			\begin{cjsolution}
				\cjwhy{\(\tan(\alpha) - \tan(\beta) = \frac{\sin(\alpha)\cos(\beta) - \cos(\alpha)\sin(\beta)}{\cos(\alpha)\cos(\beta)}\)}{\(\sin(a - b) = \sin(a)\cos(b) - \cos(a)\sin(b)\)}
				\cjcontinue{\(\tan(\alpha) - \tan(\beta) = \frac{\sin(\alpha)\cos(\beta)}{\cos(\alpha)\cos(\beta)} - \frac{\cos(\alpha)\sin(\beta)}{\cos(\alpha)\cos(\beta)}\)}
				\cjcontinue{\(\tan(\alpha) - \tan(\beta) = \frac{\sin(\alpha)}{\cos(\alpha)} - \frac{\sin(\beta)}{\cos(\beta)}\)}
				\cjfastep{\(\tan(\alpha) - \tan(\beta) = \tan(\alpha) - \tan(\beta)\)}{\(\tan(\theta) = \frac{\sin(\theta)}{\cos(\theta)}\)}
			\end{cjsolution}

		\cjitem{\(\csc(x + y) = \frac{\csc(x)\csc(y)}{\csc(x)\cos(x) + \csc(y)\cos(y)}\)}
			\begin{cjsolution}
				\cjwhy{\(\csc(x + y) = \frac{(\frac{1}{\sin(x)})(\frac{1}{\sin(y)})}{(\frac{1}{\sin(x)})\cos(x) + (\frac{1}{\sin(y)})\cos(y)}\)}{\(\csc(\theta) = \frac{1}{\sin(\theta)}\)}
				\cjcontinue{\(\csc(x + y) = \frac{\frac{1}{\sin(x)\sin(y)}}{\frac{\cos(x)}{\sin(x)} + \frac{\cos(y)}{\sin(y)}}\)}
				\cjcontinue{\(\csc(x + y) = \frac{\frac{1}{\sin(x)\sin(y)}}{\frac{\cos(x)\sin(y)}{\sin(x)\sin(y)} + \frac{sin(x)\cos(y)}{\sin(x)\sin(y)}}\)}
				\cjcontinue{\(\csc(x + y) = \frac{\frac{1}{\sin(x)\sin(y)}}{\frac{\sin(x)\cos(y) + \cos(x)\sin(y)}{\sin(x)\sin(y)}}\)}
				\cjcontinue{\(\csc(x + y) = \frac{1}{\sin(x)\cos(y) + \cos(x)\sin(y)}\)}
				\cjsubwhy{\(\csc(x + y) = \frac{1}{\sin(x + y)}\)}{\(\sin(a + b) = \sin(a)\cos(b) + \cos(a)\sin(b)\)}
				\cjfastep{\(\csc(x + y) = \csc(x + y)\)}{\(\csc(\theta) = \frac{1}{\sin(\theta)}\)}
			\end{cjsolution}

		\cjitem{\(\frac{\sin(2\theta) + \sin(\theta)}{\cos(2\theta) + \cos(\theta) + 1} = \tan(\theta)\)}
			\begin{cjsolution}
				\cjwhy{\(\tan(\theta) = \frac{2\sin(\theta)\cos(\theta) + \sin(\theta)}{\cos(2\theta) + \cos(\theta) + 1}\)}{\(\sin(2\theta) = 2\sin(\theta)\cos(\theta)\)}
				\cjsubwhy{\(\tan(\theta) = \frac{2\sin(\theta)\cos(\theta) + \sin(\theta)}{2\cos^2(\theta) - 1 + \cos(\theta) + 1}\)}{\(\cos(2\theta) = 2\cos^2(\theta) - 1\)}
				\cjcontinue{\(\tan(\theta) = \frac{2\sin(\theta)\cos(\theta) + \sin(\theta)}{2\cos^2(\theta) + \cos(\theta)}\)}
				\cjcontinue{\(\tan(\theta) = \frac{\sin(\theta)(2\cos(\theta) + 1)}{\cos(\theta)(2\cos(\theta) + 1)}\)}
				\cjcontinue{\(\tan(\theta) = \frac{\sin(\theta)}{\cos(\theta)}\)}
				\cjfastep{\(\tan(\theta) = \tan(\theta)\)}{\(\tan(\theta) = \frac{\sin(\theta)}{\cos(\theta)}\)}
			\end{cjsolution}

		\cjitem{\(\sec(2x) = \frac{\tan(x) + \cot(x)}{\cot(x) - \tan(x)}\)}
			\begin{cjsolution}
				\cjwhy{\(\sec(2x) = \frac{\frac{\sin(x)}{\cos(x)} + \cot(x)}{\cot(x) - \frac{\sin(x)}{\cos(x)}}\)}{\(\tan(\theta) = \frac{\sin(\theta)}{\cos(\theta)}\)}
				\cjsubwhy{\(\sec(2x) = \frac{\frac{\sin(x)}{\cos(x)} + \frac{\cos(x)}{\sin(x)}}{\frac{\cos(x)}{\sin(x)} - \frac{\sin(x)}{\cos(x)}}\)}{\(\cot(\theta) = \frac{\cos(\theta)}{\sin(\theta)}\)}
				\cjcontinue{\(\sec(2x) = \frac{\frac{\sin^2(x)}{\sin(x)\cos(x)} + \frac{\cos^2(x)}{\sin(x)\cos(x)}}{\frac{\cos^2(x)}{\sin(x)\cos(x)} - \frac{\sin^2(x)}{\sin(x)\cos(x)}}\)}
				\cjcontinue{\(\sec(2x) = \frac{\frac{\sin^2(x) + \cos^2(x)}{\sin(x)\cos(x)}}{\frac{\cos^2(x) - \sin^2(x)}{\sin(x)\cos(x)}}\)}
				\cjcontinue{\(\sec(2x) = \frac{\sin^2(x) + \cos^2(x)}{\cos^2(x) - \sin^2(x)}\)}
				\cjsubwhy{\(\sec(2x) = \frac{1}{\cos^2(x) - \sin^2(x)}\)}{\(\sin^2(\theta) + \cos^2(\theta) = 1\)}
				\cjsubwhy{\(\sec(2x) = \frac{1}{\cos(2x)}\)}{\(\cos(2\theta) = \cos^2(\theta) - \sin^2(\theta)\)}
				\cjfastep{\(\sec(2x) = \sec(2x)\)}{\(\sec(\theta) = \frac{1}{\cos(\theta)}\)}
			\end{cjsolution}

		\cjitem{\(\csc(2\beta) - \cot(2\beta) = \tan(\beta)\)}
			\begin{cjsolution}
				\cjwhy{\(\tan(\beta) = \frac{1}{\sin(2\beta)} - \cot(2\beta)\)}{\(\csc(\theta) = \frac{1}{\sin(\theta)}\)}
				\cjsubwhy{\(\tan(\beta) = \frac{1}{\sin(2\beta)} - \frac{\cos(2\beta)}{\sin(2\beta)}\)}{\(\cot(\theta) = \frac{\cos(\theta)}{\sin(\theta)}\)}
				\cjcontinue{\(\tan(\beta) = \frac{1 - \cos(2\beta)}{\sin(2\beta)}\)}
				\cjsubwhy{\(\tan(\beta) = \frac{1 - (\cos^2(\beta) - \sin^2(\beta))}{\sin(2\beta)}\)}{\(\cos(2\theta) = \cos^2(\theta) - \sin^2(\theta)\)}
				\cjcontinue{\(\tan(\beta) = \frac{1 - \cos^2(\beta) + \sin^2(\beta)}{\sin(2\beta)}\)}
				\cjsubwhy{\(\tan(\beta) = \frac{1 - \cos^2(\beta) + \sin^2(\beta)}{2\sin(\beta)\cos(\beta)}\)}{\(\sin(2\theta) = 2\sin(\theta)\cos(\theta)\)}
				\cjsubwhy{\(\tan(\beta) = \frac{\sin^2(\beta) + \sin^2(\beta)}{2\sin(\beta)\cos(\beta)}\)}{\(\sin^2(\theta) + \cos^2(\theta) = 1\)}
				\cjcontinue{\(\tan(\beta) = \frac{2\sin^2(\beta)}{2\sin(\beta)\cos(\beta)}\)}
				\cjcontinue{\(\tan(\beta) = \frac{\sin(\beta)}{\cos(\beta)}\)}
				\cjfastep{\(\tan(\beta) = \tan(\beta)\)}{\(\tan(\theta) = \frac{\sin(\theta)}{\cos(\theta)}\)}
			\end{cjsolution}
	\end{cjsection}
